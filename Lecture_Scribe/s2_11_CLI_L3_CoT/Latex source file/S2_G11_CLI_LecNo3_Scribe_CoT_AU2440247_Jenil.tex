\begin{itemize}

    \item Total milestones: M1 -- M6

\end{itemize}

\subsection{Project Milestones}
\begin{enumerate}
    \item Concept Evolution Maps
    \item Mathematical Modelling
    \item Coding: Simulation and Computation
    \item Inference and Randomized Algorithms
    \item Randomized Algorithm Application
    \item Derivation of Bounds, Analysis, and Final Compilation
\end{enumerate}
