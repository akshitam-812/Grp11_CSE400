\subsection*{M1: Kickstart}
\begin{itemize}

    \item Area identification
    \item Background
    \item Motivation
    \item Problem formulation

\subsection*{M2: Mathematical Modeling}
\begin{itemize}
    \item Mathematical modelling of the selected problem in \textbf{any domain}
    \item Topics include:
    \begin{itemize}
        \item Random Variables (RV)
        \item PMF / PDF
        \item CDF
        \item Multivariate Random Variables
        \item Joint PMF / PDF / CDF
    \end{itemize}
\end{itemize}

\subsection*{M3: Coding}
\begin{itemize}
    \item Simulation
    \item Computation
\end{itemize}

\subsection*{M4: Inference}
\begin{itemize}
    \item Choose a randomized algorithm
    \item Understand the algorithm
    \item Code the algorithm
\end{itemize}

\subsection*{M5: Randomized Algorithms}
\begin{itemize}
    \item Apply the randomized algorithm to the domain problem
    \item Obtain results
    \item New inferences in comparison to deterministic algorithms
\end{itemize}

\subsection*{M6: Derivation and Analysis}
\begin{itemize}
    \item Derive bounds
    \item Perform analysis
    \item Compile and submit final deliverables
\end{itemize}
