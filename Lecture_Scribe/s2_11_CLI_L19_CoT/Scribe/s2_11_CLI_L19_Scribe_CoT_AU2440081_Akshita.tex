\section{Joint Distribution of N Random Variables}

\subsection{Definition}

\begin{itemize}
    \item Consider $N$ random variables:
    \[
    X_1, X_2, \dots, X_N
    \]

    \item The \textbf{joint distribution} characterizes the probability behavior of all variables simultaneously.
\end{itemize}

\subsection{Joint PMF, CDF, and PDF}

\subsubsection{Joint PMF}

\begin{itemize}
    \item Defined for discrete random variables:
    \[
    P(X_1 = x_1, X_2 = x_2, \dots, X_N = x_N)
    \]
\end{itemize}

\subsubsection{Joint CDF}

\begin{itemize}
    \item Defined as:
    \[
    F_{X_1, X_2, \dots, X_N}(x_1, x_2, \dots, x_N)
    \]
\end{itemize}

\subsubsection{Joint PDF}

\begin{itemize}
    \item Defined for continuous random variables:
    \[
    f_{X_1, X_2, \dots, X_N}(x_1, x_2, \dots, x_N)
    \]
\end{itemize}

\subsection{Independence of Random Variables}

\begin{itemize}
    \item Random variables $(X_1, X_2, \dots, X_N)$ are independent if their joint distribution satisfies independence conditions.
\end{itemize}
